  477.   /**

  481.    * match if you give it an abbreviation of a name or alias.
  482.    *
  483.    * @param  string $name A command name or a command alias

  487.   public function findCommand($name)

  490.     $namespace = '';

  494.       $name = substr($name, $pos + 1);

  496.  
  497.     $fullName = $namespace ? $namespace.':'.$name : $name;
  498.  
  499.     // name
  500.     $commands = array();
  501.     foreach ($this->commands as $command)
  502.     {
  503.       if ($command->getNamespace() == $namespace)
  504.       {
  505.         $commands[] = $command->getName();

  508.  
  509.     $abbrevs = static::getAbbreviations($commands);
  510.     if (isset($abbrevs[$name]) && 1 == count($abbrevs[$name]))
  511.     {
  512.       return $this->getCommand($namespace ? $namespace.':'.$abbrevs[$name][0] : $abbrevs[$name][0]);
  513.     }
  514.  
  515.     if (isset($abbrevs[$name]) && count($abbrevs[$name]) > 1)
  516.     {
  517.       $suggestions = $this->getAbbreviationSuggestions(array_map(function ($command) use ($namespace) { return $namespace.':'.$command; }, $abbrevs[$name]));
  518.  
  519.       throw new \InvalidArgumentException(sprintf('Command "%s" is ambiguous (%s).', $fullName, $suggestions));
  520.     }
  521.  
  522.     // aliases
  523.     $abbrevs = static::getAbbreviations(array_keys($this->aliases));
  524.     if (!isset($abbrevs[$fullName]))
  525.     {
  526.       throw new \InvalidArgumentException(sprintf('Command "%s" is not defined.', $fullName));
  527.     }
  528.  
  529.     if (count($abbrevs[$fullName]) > 1)
  530.     {
  531.       throw new \InvalidArgumentException(sprintf('Command "%s" is ambiguous (%s).', $fullName, $this->getAbbreviationSuggestions($abbrevs[$fullName])));
  532.     }
  533.  
  534.     return $this->getCommand($abbrevs[$fullName][0]);
  535.   }
