  166.     /**
  167.      * Adds a relation to the focused list.
  168.      * 
  169.      * Adds a relation between all the focused entities specified in the
  170.      * current expression and the passed $rel entity
  171.      * 
  172.      * @param MindEntity &$rel The entity to be added
  173.      * @param String $linkType The type of like(action, possibility or must)
  174.      * @param String $linkVerb The verb used to identify the relation

  176.      * @param Mixed $max The maximun quantifier
  177.      * @param boolean [$uniqueRef] If it should be an unique referation

  180.     public static function &addRelationToFocused(MindEntity &$rel, $linkType,
  181.                                                 $linkVerb, $min, $max,
  182.                                                 $uniqueRef=false)
  183.     {
  184.         $arRet= Array();
  185.         // for each focused entity
  186.         foreach(self::$focused as &$focus)
  187.         {
  188.             /*
  189.              * we will use this relationName as index on an
  190.              * indexed array to speed up the search for
  191.              * relations in the future

  193.             $relationName= $focus->name."_".$rel->name;
  194.  
  195.             // let's create the relation itself
  196.             $curRelation= new MindRelation($relationName);
  197.  
  198.             $curRelation->setLinkType($linkType)
  199.                         ->setMin($min)
  200.                         ->setMax($max)
  201.                         ->setUsedVerb($linkVerb)
  202.                         ->setEntities(
  203.                                 self::$entities[$focus->name],
  204.                                 self::$entities[$rel->name]);
  205.             // now, both entities will POINT to the same relation
  206.             $focus->addRef($curRelation);

  209.             
  210.             // and let's use the relation name as index, as said before
  211.             self::$relations[$relationName]= $curRelation;
  212.             $arRet[]= &$curRelation;
  213.         }
  214.         return $arRet;
  215.     }

  244.     /**
  245.      * Analyzes the project.
  246.      * 
  247.      * This method receives each expression and analizes
  248.      * the best way to act and to store the understood
  249.      * structure
  250.      * 
  251.      * @param String $expression 
  252.      * @param String $structure 
  253.      * @param Array $structureKeys 
  254.      * @return Boolean True if everything went ok, false when any error occurred
  255.      */
  256.     public static function analize($expression, $structure, Array $structureKeys){
  257.         // I'm gonna try to put it in stepByStep style, to
  258.         // get it easier for me and for you to understand
  259.         // and follow the thoughts
  260.  
  261.         // setting up
  262.         self::clearFocused();
  263.         $tmpProperties= Array();
  264.         $i            = 0;
  265.         $linkVerb     = null;
  266.         $min          = null;
  267.         $max          = null;
  268.         $linkType     = 'action';
  269.         $relation     = false;
  270.         $posVerb      = false;
  271.  
  272.         // foreach token
  273.         foreach($structureKeys as $token)
  274.         {
  275.             $word= $expression[$i];
  276.             $i++;
  277.             // storing the current used verb
  278.             if($token==Tokenizer::MT_VERB)
  279.             {
  280.                 $linkVerb= $word;
  281.                 $posVerb= true;
  282.                 continue;
  283.             }
  284.  
  285.             // setting quantifiers
  286.             if($token == Tokenizer::MT_NONE)
  287.             {
  288.                 $min= 0;
  289.                 continue;
  290.             }
  291.             if($token == Tokenizer::MT_ONE || $token == Tokenizer::MT_MANY)
  292.             {
  293.                 if(!is_null($max))
  294.                     $min= $max;
  295.                 $max= ($token == Tokenizer::MT_ONE)? 1: 'n';
  296.             }
  297.  
  298.             // verifying the way the entities will be linked
  299.             if($token==Tokenizer::MT_QMAY)
  300.             {
  301.                 $linkType= 'possibility';
  302.                 continue;
  303.             }
  304.             if($token==Tokenizer::MT_QMUST)
  305.             {
  306.                 $linkType= 'must';
  307.                 continue;
  308.             }
  309.  
  310.             // if it is a substantive
  311.             if($token==Tokenizer::MT_SUBST)
  312.             {
  313.                 // if it is an entity
  314.                 if(MindEntity::isEntity($word))
  315.                 {
  316.                     // fixing any special char
  317.                     $word= Mind::$lexer->fixWordChars($word);
  318.  
  319.                     // yeah, I know it looks crazy, but try to follow my thoughts
  320.                     // let's instantiate a new Entity in case it has not been
  321.                     // instantiated before
  322.                     if(!isset(self::$entities[$word]))
  323.                         self::$entities[$word]= new MindEntity($word);
  324.  
  325.                     // each instruction *should* have one focused entity
  326.                     // we will use the first entity on each expression as focus
  327.                     if(!$posVerb)
  328.                     {
  329.                         self::addToFocus(self::$entities[$word]);
  330.                     }else{
  331.                             $relation= true;
  332.                             // if min or max quantifier have not been set
  333.                             if(is_null($min))
  334.                                 $min= ($linkType == 'must')? 1: 0;
  335.                             if(is_null($max))
  336.                                 $max= 'n';
  337.  
  338.                             /*
  339.                              * here, if it is an entity and the focused
  340.                              * entities have already been selected(post verb),
  341.                              * it means it is the second entity on the
  342.                              * instruction, so, it is a relation between entities
  343.                              */
  344.                             
  345.                             self::addRelationToFocused(self::$entities[$word],
  346.                                                        $linkType,
  347.                                                        $linkVerb,
  348.                                                        $min,
  349.                                                        $max);
  350.                          }
  351.                 }else{
  352.                     // ok, after that, this is just easy :)
  353.                     // let's store all the properties to a temporary array
  354.                     $tmpProperties[]= new MindProperty($word);
  355.                 }
  356.             }
  357.         }
  358.         
  359.         // adding the properties to the focused entity
  360.         // we're doing it now, because according to the selected idiom
  361.         // the sequence of focused entity and properties may vary
  362.         if(sizeof($tmpProperties)>0 && $posVerb)
  363.         {
  364.             foreach($tmpProperties as $prop)
  365.                 self::addPropertyToFocused($prop);
  366.         }
  367.         
  368.         // if there was a relation, we will return some details about it
  369.         if($relation && isset(self::$entities[$word]))
  370.             return Array('min'        =>$min,
  371.                          'max'        =>$max,
  372.                          'linkVerb'    =>$linkVerb,
  373.                          'linkType'    =>$linkType,
  374.                          'focus'    =>implode(', ', self::getFocusedNames()),
  375.                          'rel'        =>self::$entities[$word]->name);
  376.         // otherwise, we return the focused entities
  377.         return self::$focused;
  378.     }

  380.     /**
  381.      * Sweeps through all the matched expressions.

  384.     public static function sweep($matches)
  385.     {
  386.         // let's clear the Analyst memory as it uses static properties
  387.         self::reset();

  391.         {
  392.             $len= strlen($found[0]);
  393.             $expression= array_slice(Token::$words, $found[1], $len);
  394.             $tokens= array_slice(Token::$spine, $found[1], $len);
  395.             $struct= $found[0];
  396.  
  397.             // let's analize it, now
  398.             // Analyst will store it on its own static structure
  399.             Analyst::analize($expression, $struct, $tokens);
  400.         }
  401.         self::clearFocused();
  402.         self::normalizeIt();
  403.     }
